Stack

5.1.1 Runtime Stack

  • 使用 ESP 這個暫存器
  • 很少直接操作，通常通過指令例如 CALL、Ret、Push、pop

5.1.2 PUSH and POP Instructions

  1. push: 加入值
  2. pop: 取出值
  3. Pushfd、Popfd: 操作 [EFLAGS] 暫存器
  4. Pushad、Popa: 放 暫存器
  5. PUSHA、POPA: 16位元 時使用

範例: 反轉字串


; 反轉字串 (RevStr.asm)
.386
.model flat,stdcall
.stack 4096
ExitProcess PROTO,dwExitCode:DWORD

.data
 aName BYTE "Abraham Lincoln",0
 nameSize = ($ - aName) - 1

.code
main PROC

 ; 將字串推入堆疊
 mov ecx,nameSize
 mov esi,0

L1: 
    movzx eax,aName[esi] ; 獲取字元
    push eax ; 推入堆疊
    inc esi
    loop L1

 ; 反向彈出字元並存回字串
 mov ecx,nameSize
 mov esi,0

L2: 
    pop eax ; 獲取字元
    mov aName[esi],al ; 存回字串
    inc esi
    loop L2

INVOKE ExitProcess,0
main ENDP
END main
